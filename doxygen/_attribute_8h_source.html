218template <class DependentType, class... DependencyTypes>
219class AttributeUpdateTask
220 : public AttributeUpdateTaskBase<DependentType>,
221 public SharedFactory<
222 AttributeUpdateTask<DependentType, DependencyTypes...>> {
223
224public:
225 using Actor =
226 std::function<void(std::shared_ptr<DependentType> &,
227 typename Attribute<DependencyTypes>::Ptr...)>;
228
229protected:
230 std::tuple<typename Attribute<DependencyTypes>::Ptr...> mDependencies;
231 Actor mActorFunction;
232 UpdateTaskKind mKind;
233
234public:
235 AttributeUpdateTask(UpdateTaskKind kind, Actor &actorFunction,
236 typename Attribute<DependencyTypes>::Ptr... dependencies)
237 : mDependencies(std::forward<typename Attribute<DependencyTypes>::Ptr>(
238 dependencies)...),
239 mActorFunction(actorFunction), mKind(kind) {}
240
241 virtual void
242 executeUpdate(std::shared_ptr<DependentType> &dependent) override {
243 mActorFunction(
244 dependent,
245 std::get<typename Attribute<DependencyTypes>::Ptr...>(mDependencies));
246 }
247
251 virtual AttributeBase::List getDependencies() override {
252 return std::apply(
253 [](auto &&...elems) {
254 return std::vector<AttributeBase::Ptr>{
255 std::forward<decltype(elems)>(elems)...};
256 },
257 mDependencies);
258 };

372 template <class U>
373 typename Attribute<U>::Ptr
374 derive(typename AttributeUpdateTask<U, T>::Actor getter =
375 typename AttributeUpdateTask<U, T>::Actor(),
376 typename AttributeUpdateTask<U, T>::Actor setter =
377 typename AttributeUpdateTask<U, T>::Actor()) {
378 auto derivedAttribute = std::make_shared<AttributeDynamic<U>>();
379 if (setter) {
380 derivedAttribute->addTask(
381 UpdateTaskKind::UPDATE_ON_SET,
382 AttributeUpdateTask<U, T>::make(
383 UpdateTaskKind::UPDATE_ON_SET, setter,
384 Attribute<T>::Ptr(this->shared_from_this())));
385 }
386 if (getter) {
387 derivedAttribute->addTask(
388 UpdateTaskKind::UPDATE_ON_GET,
389 AttributeUpdateTask<U, T>::make(
390 UpdateTaskKind::UPDATE_ON_GET, getter,
391 Attribute<T>::Ptr(this->shared_from_this())));
392 }
393 return derivedAttribute;
394 }

574template <class T>
575class AttributeDynamic : public Attribute<T>,
576 public SharedFactory<AttributeDynamic<T>> {
577 friend class SharedFactory<AttributeDynamic<T>>;
578
579protected:
580 std::vector<typename AttributeUpdateTaskBase<T>::Ptr> updateTasksOnce;
581 std::vector<typename AttributeUpdateTaskBase<T>::Ptr> updateTasksOnGet;
582 std::vector<typename AttributeUpdateTaskBase<T>::Ptr> updateTasksOnSet;
583
584public:
585 AttributeDynamic(T initialValue = T()) : Attribute<T>(initialValue) {}
586
592 void addTask(UpdateTaskKind kind,
593 typename AttributeUpdateTaskBase<T>::Ptr task) {
594 switch (kind) {
595 case UpdateTaskKind::UPDATE_ONCE:
596 updateTasksOnce.push_back(task);
597 // THISISBAD: This is probably not the right time to run this kind of task
598 task->executeUpdate(this->mData);
599 break;
600 case UpdateTaskKind::UPDATE_ON_GET:
601 updateTasksOnGet.push_back(task);
602 break;
603 case UpdateTaskKind::UPDATE_ON_SET:
604 updateTasksOnSet.push_back(task);
605 break;
606 case UpdateTaskKind::UPDATE_ON_SIMULATION_STEP:
607 throw InvalidArgumentException();
608 };
609 }
610
615 void clearTasks(UpdateTaskKind kind) {
616 switch (kind) {
617 case UpdateTaskKind::UPDATE_ONCE:
618 updateTasksOnce.clear();
619 break;
620 case UpdateTaskKind::UPDATE_ON_GET:
621 updateTasksOnGet.clear();
622 break;
623 case UpdateTaskKind::UPDATE_ON_SET:
624 updateTasksOnSet.clear();
625 break;
626 case UpdateTaskKind::UPDATE_ON_SIMULATION_STEP:
627 throw InvalidArgumentException();
628 };
629 }
630
634 void clearAllTasks() {
635 updateTasksOnce.clear();
636 updateTasksOnGet.clear();
637 updateTasksOnSet.clear();
638 }
639
640 virtual void setReference(typename Attribute<T>::Ptr reference) override {
641 typename AttributeUpdateTask<T, T>::Actor getter =
642 [](std::shared_ptr<T> &dependent,
643 typename Attribute<T>::Ptr dependency) {
644 dependent = dependency->asRawPointer();
645 };
646 this->clearAllTasks();
647 if (reference->isStatic()) {
648 this->addTask(UpdateTaskKind::UPDATE_ONCE,
649 AttributeUpdateTask<T, T>::make(UpdateTaskKind::UPDATE_ONCE,
650 getter, reference));
651 } else {
652 this->addTask(UpdateTaskKind::UPDATE_ON_GET,
653 AttributeUpdateTask<T, T>::make(
654 UpdateTaskKind::UPDATE_ON_GET, getter, reference));
655 }
656 }
657
658 virtual std::shared_ptr<T> asRawPointer() override {
659 for (typename AttributeUpdateTaskBase<T>::Ptr task : updateTasksOnGet) {
660 task->executeUpdate(this->mData);
661 }
662 return this->mData;
663 }
664
665 virtual void set(T value) override {
666 *this->mData = value;
667 for (typename AttributeUpdateTaskBase<T>::Ptr task : updateTasksOnSet) {
668 task->executeUpdate(this->mData);
669 }
670 };
671
672 virtual T &get() override {
673 for (typename AttributeUpdateTaskBase<T>::Ptr task : updateTasksOnGet) {
674 task->executeUpdate(this->mData);
675 }
676 return *this->mData;
677 };
678
679 virtual bool isStatic() const override { return false; }
680
686 virtual void appendDependencies(AttributeBase::Set *deps) override {
687 deps->insert(this->shared_from_this());
688
689 AttributeBase::Set newDeps = AttributeBase::Set();
690 for (typename AttributeUpdateTaskBase<T>::Ptr task : updateTasksOnce) {
691 AttributeBase::List taskDeps = task->getDependencies();
692 newDeps.insert(taskDeps.begin(), taskDeps.end());
693 }
694
695 for (typename AttributeUpdateTaskBase<T>::Ptr task : updateTasksOnGet) {
696 AttributeBase::List taskDeps = task->getDependencies();
697 newDeps.insert(taskDeps.begin(), taskDeps.end());
698 }
699
700 for (auto dependency : newDeps) {
701 dependency->appendDependencies(deps);
702 }
703 }
704};
